async function showProfileModal(staff) { currentProfileStaff = staff; const modal = document.createElement('div'); modal.className = 'profile-modal'; modal.innerHTML = `

👤 Staff Profile

${escapeHtml(staff.name)} (${staff.role})

Verify current password first, then draw your new pattern

Draw your NEW pattern (at least 3 dots):

${[1,2,3,4,5,6,7,8,9].map(i => `
`).join('')}
Draw your pattern...
`; document.body.appendChild(modal); // ---------- Password change logic ---------- const updateBtn = modal.querySelector('#updatePasswordBtn'); const currentPassInput = modal.querySelector('#currentPass'); const newPassInput = modal.querySelector('#newPass'); const confirmPassInput = modal.querySelector('#confirmPass'); const passwordMsg = modal.querySelector('#passwordMessage'); updateBtn.onclick = async () => { const currentPass = currentPassInput.value; const newPass = newPassInput.value; const confirmPass = confirmPassInput.value; if (!currentPass || !newPass || !confirmPass) { passwordMsg.textContent = '❌ Please fill all fields'; passwordMsg.className = 'profile-message error'; return; } if (newPass !== confirmPass) { passwordMsg.textContent = '❌ New passwords do not match'; passwordMsg.className = 'profile-message error'; return; } if (newPass.length < 4) { passwordMsg.textContent = '❌ Password must be at least 4 digits'; passwordMsg.className = 'profile-message error'; return; } if (currentPass !== staff.pass) { passwordMsg.textContent = '❌ Current password is incorrect'; passwordMsg.className = 'profile-message error'; return; } const success = await saveUserCredentials(staff.id, { pass: newPass, pattern: staff.pattern }); if (success) { staff.pass = newPass; passwordMsg.textContent = '✅ Password updated successfully!'; passwordMsg.className = 'profile-message success'; currentPassInput.value = ''; newPassInput.value = ''; confirmPassInput.value = ''; showMessage(`Password updated for ${staff.name}`); } else { passwordMsg.textContent = '❌ Failed to update password'; passwordMsg.className = 'profile-message error'; console.error('saveUserCredentials returned false'); } }; // ---------- Pattern change logic ---------- const tabs = modal.querySelectorAll('.profile-tab'); const passwordSection = modal.querySelector('#passwordSection'); const patternSection = modal.querySelector('#patternSection'); tabs.forEach(tab => { tab.onclick = () => { tabs.forEach(t => t.classList.remove('active')); tab.classList.add('active'); if (tab.dataset.tab === 'password') { passwordSection.style.display = 'block'; patternSection.style.display = 'none'; } else { passwordSection.style.display = 'none'; patternSection.style.display = 'block'; } }; }); const verifyPatternBtn = modal.querySelector('#verifyForPatternBtn'); const patternCurrentPass = modal.querySelector('#patternCurrentPass'); const patternChangeArea = modal.querySelector('#patternChangeArea'); const patternMsg = modal.querySelector('#patternMessage'); const savePatternBtn = modal.querySelector('#savePatternBtn'); const newPatternGrid = modal.querySelector('#newPatternGrid'); const newPatternPreview = modal.querySelector('#newPatternPreview'); const newDots = modal.querySelectorAll('#newPatternGrid .pattern-dot'); let newPatternSequence = []; let isDrawingNew = false; let patternDrawListenersAttached = false; function resetNewPatternDrawing() { newPatternSequence = []; newDots.forEach(dot => dot.classList.remove('selected')); newPatternPreview.textContent = 'Draw your pattern...'; } function updateNewPatternPreview() { if (newPatternSequence.length === 0) { newPatternPreview.textContent = 'Draw your pattern...'; } else { newPatternPreview.textContent = newPatternSequence.map(n => '●').join(' → '); } } function getNewDotFromEvent(e) { const rect = newPatternGrid.getBoundingClientRect(); let clientX, clientY; if (e.touches) { clientX = e.touches[0].clientX; clientY = e.touches[0].clientY; } else { clientX = e.clientX; clientY = e.clientY; } for (let dot of newDots) { const dotRect = dot.getBoundingClientRect(); if (clientX >= dotRect.left && clientX <= dotRect.right && clientY >= dotRect.top && clientY <= dotRect.bottom) { return { dotNum: parseInt(dot.dataset.num), element: dot }; } } return null; } function startNewDraw(e) { e.preventDefault(); isDrawingNew = true; const dotInfo = getNewDotFromEvent(e); if (dotInfo && !newPatternSequence.includes(dotInfo.dotNum)) { newPatternSequence = [dotInfo.dotNum]; dotInfo.element.classList.add('selected'); updateNewPatternPreview(); } } function drawNew(e) { e.preventDefault(); if (!isDrawingNew) return; const dotInfo = getNewDotFromEvent(e); if (dotInfo && !newPatternSequence.includes(dotInfo.dotNum)) { newPatternSequence.push(dotInfo.dotNum); dotInfo.element.classList.add('selected'); updateNewPatternPreview(); } } function endNewDraw(e) { e.preventDefault(); isDrawingNew = false; } function cleanupNewPatternDrawListeners() { if (patternDrawListenersAttached) { window.removeEventListener('mousemove', drawNew); window.removeEventListener('mouseup', endNewDraw); window.removeEventListener('touchmove', drawNew); window.removeEventListener('touchend', endNewDraw); patternDrawListenersAttached = false; } } verifyPatternBtn.onclick = async () => { const currentPass = patternCurrentPass.value; if (!currentPass) { patternMsg.textContent = '❌ Please enter current password'; patternMsg.className = 'profile-message error'; return; } if (currentPass !== staff.pass) { patternMsg.textContent = '❌ Incorrect password'; patternMsg.className = 'profile-message error'; return; } patternMsg.textContent = '✅ Password verified! Draw your new pattern below'; patternMsg.className = 'profile-message success'; patternChangeArea.style.display = 'block'; verifyPatternBtn.disabled = true; verifyPatternBtn.style.opacity = '0.5'; resetNewPatternDrawing(); // Remove any previous listeners cleanupNewPatternDrawListeners(); // Attach new listeners window.addEventListener('mousemove', drawNew); window.addEventListener('mouseup', endNewDraw); window.addEventListener('touchmove', drawNew); window.addEventListener('touchend', endNewDraw); patternDrawListenersAttached = true; newPatternGrid.addEventListener('mousedown', startNewDraw); newPatternGrid.addEventListener('touchstart', startNewDraw, { passive: false }); }; savePatternBtn.onclick = async () => { if (newPatternSequence.length < 3) { patternMsg.textContent = '❌ Pattern must have at least 3 dots'; patternMsg.className = 'profile-message error'; return; } const newPattern = patternToString(newPatternSequence); console.log(`Saving new pattern for ${staff.id}: ${newPattern}`); try { const success = await saveUserCredentials(staff.id, { pass: staff.pass, pattern: newPattern }); if (success) { staff.pattern = newPattern; patternMsg.textContent = '✅ Pattern updated successfully!'; patternMsg.className = 'profile-message success'; resetNewPatternDrawing(); showMessage(`Pattern updated for ${staff.name}`); // Close modal after 1.5 seconds setTimeout(() => { cleanupNewPatternDrawListeners(); modal.remove(); }, 1500); } else { patternMsg.textContent = '❌ Failed to update pattern (Firebase error)'; patternMsg.className = 'profile-message error'; console.error('saveUserCredentials returned false'); } } catch (err) { console.error('Exception saving pattern:', err); patternMsg.textContent = '❌ Error saving pattern: ' + err.message; patternMsg.className = 'profile-message error'; } }; // Clean up when modal is closed const closeModal = () => { cleanupNewPatternDrawListeners(); modal.remove(); }; modal.querySelector('#closeProfileBtn').onclick = closeModal; modal.onclick = (e) => { if (e.target === modal) closeModal(); }; }
